### 2.2 Core/Definitions（静态定义类）

#### `NodeDefinitionBase.cs`
```
功能：分类节点与物品节点共享的基础定义抽象类
      统一 ID、名称、排序、父节点等公共元数据
```
```csharp
public abstract class NodeDefinitionBase
{
    // ====== 标识 ======
    string Id;                    // 唯一标识符，全小写+下划线，如 "mat_wood_firewood"
    string ParentId;              // 父节点ID，根节点为 null 或 ""
    
    // ====== 显示信息 ======
    string NameKey;               // 名称文本（前期为中文，后期作为多语言key）
    string DescriptionKey;        // 描述文本（前期中文，后期多语言key）
    
    // ====== 排序 ======
    int DefinitionOrder;          // 定义顺位，稀疏整数(100,200,300...)，同级内排序
    
    // ====== 节点类型标记 ======
    bool IsAbstract;              // true=分类节点(Category)，不可直接持有
                                  // false=具体物品(Item)，可持有
    
    // ====== 弃用/迁移 ======
    bool Deprecated;              // 是否已弃用
    string DeprecatedSince;       // 弃用起始版本号，如 "1.2.0"
    string ReplacementId;         // 可选，弃用后替换为哪个新物品ID
    
    // ====== 可见性 ======
    bool IsVisible;               // 是否在UI中可见（默认true）
    
    // ====== 扩展槽 ======
    Dictionary<string, object> ExtensionData;  
    // 预留的通用扩展字典，用于挂载未来的附加属性
    // 例如后续可加入 "recipe_ids", "buff_id" 等
    
    // ------ 方法 ------
    // 获取本地化后的显示名称（优先查多语言表，查不到则返回NameKey本身）
    string GetDisplayName();
    // 获取本地化后的描述文本
    string GetDescription();
}
```

#### `ItemDefinition.cs`
```
功能：具体物品的静态模板定义，继承 NodeDefinitionBase
```

```csharp
public class ItemDefinition : NodeDefinitionBase
{
    // ====== 稀有度 ======
    Rarity BaseRarity;            // 基础稀有度，默认 Rarity.Common
    
    // ====== 数量上限 ======
    int BaseMaxStack;             // 基础持有上限（如999），扩容由玩家态管理
    
    // ====== 价格 ======
    int BuyPrice;                 // 商店购买价格（0表示不可购买）
    int SellPrice;                // 玩家出售价格（恒定价格）
    
    // ====== 标签 ======
    ItemTag Tags;                 // 物品类型标签（可按位组合，用于筛选）
    
    // ====== 品质与耐久（装备/工具用）======
    bool HasDurability;           // 是否具有耐久度属性
    int MaxDurability;            // 最大耐久度（HasDurability为true时有效）
    
    // ====== 配方信息（合成系统预留）======
    // 配方数据存于独立配置，此处仅记录关联的配方ID列表
    List<string> RecipeIds;       // 该物品参与的合成配方ID列表
    
    // ====== 获取来源提示 ======
    string AcquisitionHintKey;    // 获取途径提示文本key（如"通过砍树获得"）
    
    // ====== 详细描述（字典界面）======
    string DetailDescriptionKey;  // 物品详细信息文本key（字典面板中展示）
    
    // ====== 工具/装备效果（扩展接口）======
    // 预留：工具附带的加成属性，如 "原木获取率+100%"
    // 使用通用字典或专用结构
```

#### `SkillLevelEntry.cs`
```
功能：技能等级表中单条等级的数据
```
```csharp
public class SkillLevelEntry
{
    int Level;                    // 当前等级
    long ExpToNextLevel;          // 升到下一级所需经验值（满级时为0或-1）
    float OutputPerAction;        // 每次结算的基础产出量（如5.0个原木）
    float ActionInterval;         // 每次读条/结算的时间间隔（秒，如6.0）
    // 备注：实际产出 = OutputPerAction * 工具加成倍率
    // 备注：实际间隔 = ActionInterval * 读条速度加成系数
}
```

#### `EventDefinition.cs`
```
功能：游戏事件（按钮）的静态定义
```
```csharp
public class EventDefinition
{
    string Id;                    // 事件唯一ID，如 "evt_pickup_stone_axe"
    string NameKey;               // 事件按钮显示名key
    string DescriptionKey;        // 事件描述key
    EventType Type;               // 事件类型（一次性/可重复/挂机）
    
    // ====== 前置条件 ======
    List<EventCondition> Prerequisites;
    // 前置条件列表：满足所有条件后按钮才出现/可点击
    // EventCondition 是一个结构体，见下方说明
    
    // ====== 消耗 ======
    List<ItemCostEntry> Costs;    // 点击时消耗的物品列表（ID+数量）
    
    // ====== 奖励/效果 ======
    List<ItemRewardEntry> Rewards;// 点击后获得的物品列表（ID+数量）
    string GrantSkillExpId;       // 可选：点击后给予经验的技能ID
    float GrantSkillExpAmount;    // 可选：给予的经验值数量
    
    // ====== 挂机专用 ======
    string LinkedSkillId;         // 关联的技能ID（挂机事件用，如 "skill_chopping"）
    
    // ====== 关联 ======
    string FactionId;             // 可选：所属势力ID
    string ZoneId;                // 可选：所属区域ID
    string ButtonListGroup;       // 所属按钮列表分组（用于UI分组显示）
    
    // ====== 弃用 ======
    bool Deprecated;
    
    // ====== 扩展 ======
    Dictionary<string, object> ExtensionData;
}

// 事件前置条件结构
public class EventCondition
{
    string ConditionType;         // 条件类型："has_item", "skill_level", "faction_rep", "event_completed"
    string TargetId;              // 目标ID（物品ID/技能ID/势力ID/事件ID）
    int RequiredValue;            // 要求的数值（数量/等级/好感度）
    // 备注：可扩展为更复杂的条件表达式
}

// 物品消耗条目
public class ItemCostEntry
{
    string ItemId;                // 物品ID
    int Amount;                   // 消耗数量
}

// 物品奖励条目
public class ItemRewardEntry
{
    string ItemId;                // 物品ID
    int Amount;                   // 获得数量
    // 可选：概率掉落等后续扩展
    float DropChance;             // 掉落概率（1.0=必定获得）
}
```

```csharp
public class ItemRegistry
```

## 四、架构关键设计要点总结

| 设计原则 | 实现方式 |
|---------|---------|
| **稳定唯一ID** | 全小写下划线，不以名称为主键 |
| **配置驱动** | 所有物品/分类/技能/事件通过 YAML 声明，启动时自动注册 |
| **模板-实例分离** | Definition（静态模板）与 Stack/State（玩家动态数据）彻底分离 |
| **树形无环** | TreeValidator 提供环检测、重复ID、缺失父节点、孤立节点、顺位冲突校验 |
| **排序双层** | definition_order（定义顺位）+ player_display_order（玩家自定义排序） |
| **多级排序** | 收藏置顶 → 稀有度降序 → 定义/玩家顺位 → ID字母序兜底 |
| **事务性操作** | TransactionHelper 保证合成/交易的原子性 |
| **存档兼容** | save_version + MigrationRunner 链式升级 + deprecated 物品迁移 |
| **多语言预留** | LocalizationManager 前期返回key本身（中文），后期加载翻译表 |
| **信号解耦** | SignalBus 全局信号总线，逻辑系统与UI通过信号通信 |
| **扩展预留** | ExtensionData 字典 + 战斗/成就/装备系统仅声明框架 |
